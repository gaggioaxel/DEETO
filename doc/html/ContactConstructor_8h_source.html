364 double ContactConstructor::computeCos(PhysicalPointType a1,PhysicalPointType a2,PhysicalPointType b1,PhysicalPointType b2){
365  double l1 = a1[0] - a2[0];
366  double m1 = a1[1] - a2[1];
367  double n1 = a1[2] - a2[2];
368  double l2 = b1[0] - b2[0];
369  double m2 = b1[1] - b2[1];
370  double n2 = b1[2] - b2[2];
371  double arcos = (l1*l2 + m1*m2+n1*n2)/(sqrt(l1*l1+m1*m1+n1*n1)*sqrt(l2*l2+m2*m2+n2*n2));
372  if (arcos > 0) return arcos;
373  else return -1*arcos;
374 }
